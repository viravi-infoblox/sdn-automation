package NetMRI::SDN::Cisco;

sub getDevices {
    my $self = shift;
    
    my $api_helper = $self->getApiClient();
    my ($organizations, $msg) = $api_helper->get_organizations();
    
    my @devices;
    foreach my $org (@$organizations) {
        my ($org_devices, $error) = $api_helper->get_organization_devices($org->{id});
        
        foreach my $dev (@$org_devices) {
            my %device = (
                SdnControllerId => $api_helper->{fabric_id},
                SdnDeviceDN => "$org->{id}/$dev->{networkId}/$dev->{serial}",
                Serial => $dev->{serial},
                Model => $dev->{model},
                Vendor => $self->{vendor_name},
                IPAddress => $dev->{lanIp} || $dev->{wan1Ip},
                SWVersion => $dev->{firmware} || '',
            );
            push @devices, \%device if $device{IPAddress};
        }
    }
    return \@devices;
}
